\section{降压比计算}
\begin{frame}
  \frametitle{降压比计算}
  \begin{itemize}
    \item 谐振电感电压$V_{L_{r2}}$
      \begin{itemize}
        \item \begin{equation}
			V_{\mathrm{Lr} 2}=\frac{\frac{V_{\mathrm{in}}}{2}-V_{o}-V_{\mathrm{Cr} 2}}{1+\frac{L_{m}}{L_{r 2}} \cdot \frac{(n+1)^{2}}{n^{2}}}
			\end{equation}
      \end{itemize}
    \item 变压器励磁电感电压$V_{L_{m}}$
      \begin{itemize}
        \item \begin{equation}
			V_{\mathrm{Lm}}=\frac{L_{m}}{L_{r 2}} \cdot\left(1+\frac{1}{n}\right) \cdot V_{\mathrm{Lr} 2}
			\end{equation}
      \end{itemize}
    \item 伏秒平衡(开态)
	    \begin{itemize}
	    	\item \begin{equation}
				V_{\mathrm{Lr} 2} \cdot D+\left(-V_{\mathrm{Cr} 2}+n V_{o}\right) \cdot(1-D)=0
				\end{equation}
	    \end{itemize} 
	  \item 伏秒平衡(关态)
	    \begin{itemize}
		    \item \begin{equation}
				V_{\mathrm{Lm}} \cdot D+(1-D)\left(-n V_{o}\right)=0
				\end{equation}
      \end{itemize} 
  \end{itemize}
\end{frame}


\begin{frame}
  \frametitle{降压比计算}
  \begin{itemize}
    \item 降压比
      \begin{itemize}
        \item \begin{equation}
			\frac{V_{o}}{V_{\text {in }}}=\frac{D}{2 \cdot\left(n+1+\frac{L_{r}}{L_{m}} \cdot \frac{n^{2}}{n+1}\right)}
			\end{equation}
      \end{itemize}
  \end{itemize}
\end{frame}
